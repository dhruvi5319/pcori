```java
@Entity
@Table(name = "dashboard_configurations")
public class DashboardConfiguration {
    @Id @GeneratedValue(strategy = GenerationType.UUID)
    private UUID id;

    @Column(name = "user_id", nullable = false, unique = true)
    private UUID userId;

    @Column(nullable = false, columnDefinition = "jsonb")
    @Type(JsonBinaryType.class)  // use io.hypersistence.utils.hibernate.type.json.JsonBinaryType
    private Map<String, Object> layout = new HashMap<>();

    @Column(columnDefinition = "jsonb")
    @Type(JsonBinaryType.class)
    private Map<String, Object> widgets;

    @Column(name = "created_at") private Instant createdAt = Instant.now();
    @Column(name = "updated_at") private Instant updatedAt = Instant.now();
    @Column(name = "deleted_at") private Instant deletedAt;
}
```

NOTE: hypersistence-utils is needed for JSONB. Check if it is already in pom.xml. If NOT, add to pom.xml:
```xml
<dependency>
    <groupId>io.hypersistence</groupId>
    <artifactId>hypersistence-utils-hibernate-63</artifactId>
    <version>3.9.0</version>
</dependency>
```
Use the version that matches hibernate 6.x (Spring Boot 3.4 uses Hibernate 6.x).

Alternative if hypersistence-utils is not available: store layout as `@Column(columnDefinition = "TEXT")` String (JSON serialized/deserialized manually via ObjectMapper). Use this fallback if the maven dependency cannot be resolved, but prefer the JSONB type.

**DashboardService.java**:
```java
@Service
@RequiredArgsConstructor
@Transactional
public class DashboardService {
    private final ClassificationRepository classificationRepository;
    private final DashboardConfigurationRepository configRepository;

    @Transactional(readOnly = true)
    public DashboardMetricsDto getMetrics() {
        return computeMetrics(null, null);
    }

    @Transactional(readOnly = true)
    public DashboardMetricsDto getMetricsForRange(Instant startDate, Instant endDate) {
        return computeMetrics(startDate, endDate);
    }

    private DashboardMetricsDto computeMetrics(Instant start, Instant end) {
        // Use ClassificationRepository counts by status
        // classificationRepository.countByStatus(ClassificationStatus.X)
        // Use classificationRepository.findAll with spec or native SQL COUNT queries
        long total      = countByStatus(null, start, end);
        long classified = countByStatus(ClassificationStatus.CLASSIFIED, start, end);
        long processing = countByStatus(ClassificationStatus.PROCESSING, start, end);
        long pending    = countByStatus(ClassificationStatus.PENDING, start, end);
        long failed     = countByStatus(ClassificationStatus.FAILED, start, end);
        long needsReview = countByStatus(ClassificationStatus.NEEDS_REVIEW, start, end);
        double avgConf  = getAvgConfidence(start, end);
        return new DashboardMetricsDto(total, classified, processing, pending, failed, needsReview, avgConf);
    }

    private long countByStatus(ClassificationStatus status, Instant start, Instant end) {
        if (status == null) {
            // Total count
            if (start == null) return classificationRepository.countByDeletedAtIsNull();
            return classificationRepository.countByUploadedAtBetweenAndDeletedAtIsNull(start, end);
        }
        if (start == null) return classificationRepository.countByStatusAndDeletedAtIsNull(status);
        return classificationRepository.countByStatusAndUploadedAtBetweenAndDeletedAtIsNull(status, start, end);
    }

    private double getAvgConfidence(Instant start, Instant end) {
        // Use native query for AVG
        // Return 0.0 if no data
        return classificationRepository.findAvgConfidence(start, end);
    }

    // FR-4.5: Per-user configuration persistence
    @Transactional(readOnly = true)
    public Optional<DashboardConfigurationDto> getConfiguration(UUID userId) {
        return configRepository.findByUserIdAndDeletedAtIsNull(userId)
            .map(c -> new DashboardConfigurationDto(c.getId(), c.getUserId(),
                c.getLayout(), c.getWidgets(), c.getCreatedAt(), c.getUpdatedAt()));
    }

    public DashboardConfigurationDto saveConfiguration(UUID userId, SaveConfigurationRequest req) {
        DashboardConfiguration config = configRepository
            .findByUserIdAndDeletedAtIsNull(userId)
            .orElseGet(() -> {
                DashboardConfiguration c = new DashboardConfiguration();
                c.setUserId(userId);
                return c;
            });
        config.setLayout(req.layout() != null ? req.layout() : new HashMap<>());
        config.setWidgets(req.widgets());
        config.setUpdatedAt(Instant.now());
        DashboardConfiguration saved = configRepository.save(config);
        return new DashboardConfigurationDto(saved.getId(), saved.getUserId(),
            saved.getLayout(), saved.getWidgets(), saved.getCreatedAt(), saved.getUpdatedAt());
    }

    public void deleteConfiguration(UUID userId) {
        configRepository.findByUserIdAndDeletedAtIsNull(userId).ifPresent(c -> {
            c.setDeletedAt(Instant.now());
            configRepository.save(c);
        });
    }
}
```

```java
@RestController
@RequestMapping("/api/dashboard")
@RequiredArgsConstructor
@PreAuthorize("isAuthenticated()")
public class DashboardController {
    private final DashboardService dashboardService;

    @GetMapping("/metrics")
    public ResponseEntity<DashboardMetricsDto> getMetrics() {
        return ResponseEntity.ok(dashboardService.getMetrics());
    }

    @GetMapping("/metrics/range")
    public ResponseEntity<DashboardMetricsDto> getMetricsRange(
            @RequestParam(required = false) String startDate,
            @RequestParam(required = false) String endDate) {
        Instant start = parseInstant(startDate, Instant.now().minus(30, ChronoUnit.DAYS));
        Instant end   = parseInstant(endDate, Instant.now());
        return ResponseEntity.ok(dashboardService.getMetricsForRange(start, end));
    }

    @GetMapping("/configuration")
    public ResponseEntity<DashboardConfigurationDto> getConfiguration(Authentication auth) {
        UUID userId = getUserId(auth);
        return dashboardService.getConfiguration(userId)
            .map(ResponseEntity::ok)
            .orElse(ResponseEntity.noContent().build());
    }

    @PostMapping("/configuration")
    public ResponseEntity<DashboardConfigurationDto> createConfiguration(
            @RequestBody SaveConfigurationRequest req, Authentication auth) {
        return ResponseEntity.ok(dashboardService.saveConfiguration(getUserId(auth), req));
    }

    @PutMapping("/configuration")
    public ResponseEntity<DashboardConfigurationDto> updateConfiguration(
            @RequestBody SaveConfigurationRequest req, Authentication auth) {
        return ResponseEntity.ok(dashboardService.saveConfiguration(getUserId(auth), req));
    }

    @DeleteMapping("/configuration")
    public ResponseEntity<Void> deleteConfiguration(Authentication auth) {
        dashboardService.deleteConfiguration(getUserId(auth));
        return ResponseEntity.ok().build();
    }

    @GetMapping("/configuration/{id}")
    public ResponseEntity<DashboardConfigurationDto> getConfigurationById(
            @PathVariable UUID id, Authentication auth) {
        // Returns config only if it belongs to the authenticated user
        return dashboardService.getConfiguration(getUserId(auth))
            .filter(c -> c.id().equals(id))
            .map(ResponseEntity::ok)
            .orElse(ResponseEntity.notFound().build());
    }

    private UUID getUserId(Authentication auth) {
        User user = (User) auth.getPrincipal();
        return user.getId();
    }

    private Instant parseInstant(String value, Instant defaultValue) {
        if (value == null || value.isBlank()) return defaultValue;
        try {
            if (value.contains("T")) return Instant.parse(value);
            return LocalDate.parse(value).atStartOfDay(ZoneOffset.UTC).toInstant();
        } catch (Exception e) { return defaultValue; }
    }
}
```
